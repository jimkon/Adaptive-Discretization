tree_dims = 1
tree_size = 1000 
x = np.linspace(0, 1, 1000)
pdf = np.zeros(len(x))
pdf[int(len(x)*0.3)] = 1
pdf = pdf / np.sum(pdf)
samples = np.random.choice(x, len(x) ,p=pdf).reshape((len(x), tree_dims))
plt.plot(x, pdf)
plt.show()
plt.hist(samples)
plt.show()

tree = Tree(tree_dims, tree_size)

count = 0
tree.feed(samples)
ME = [tree.get_mean_error()]

while tree.update() and count < 20:
    tree.feed(samples)
    ME.append(tree.get_mean_error())
    count+=1
tree.plot()
print('adapted on {} updates with final ME = {} and ME decrease at {}%'.format(count,
                                                                               ME[len(ME)-1],
                                                                               100*ME[len(ME)-1]/ME[0]))
plt.plot(ME, label='ME')
plt.xlabel("Updates")
plt.legend()
plt.grid(True)
plt.show()

plot_point_density(tree)

tree_dims = 1
tree_size = 1000 
x = np.linspace(0, 1, 1000)
pdf = np.sin(3*np.pi*x)+1
pdf = np.multiply(pdf, list((1 if x_i>0.5 else 0.5) for x_i in x))
pdf = pdf / np.sum(pdf)
samples = 0.3*np.random.choice(x, len(x) ,p=pdf).reshape((len(x), tree_dims))
plt.plot(x*0.3, pdf)
plt.show()
plt.hist(samples)
plt.show()

tree = Tree(tree_dims, tree_size)

count = 0

ME = [tree.get_mean_error()]

    tree.feed(samples)
    ME.append(tree.get_mean_error())
